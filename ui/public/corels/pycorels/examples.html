Examples¶

Toy Dataset¶

import numpy as np
from corels import CorelsClassifier
X = np.array([ [1, 0, 1], [0, 1, 0], [1, 1, 1] ])
y = np.array([ 1, 0, 1])
c = CorelsClassifier(verbosity=[])
c.fit(X, y)
c.printrl()

COMPAS Dataset¶

from corels import *

X, y = load_from_csv("data/compas.csv")
c = CorelsClassifier(n_iter=10000)

a = c.fit(X, y).score(X, y)
c.save("model")
c.load("model")
print(c) # same as c.printrl(), or print(c.rl_)
